Get Started in Under 5 Minutes

npm install mcp-proxy-wrapper

import { wrapWithProxy, LLMSummarizationPlugin } from 'mcp-proxy-wrapper';

// Your existing MCP server
const server = new McpServer({ name: 'my-server', version: '1.0.0' });

// Add AI enhancement
const summaryPlugin = new LLMSummarizationPlugin();
const proxiedServer = await wrapWithProxy(server, { 
  plugins: [summaryPlugin] 
});

// Tools now have automatic AI summarization!
proxiedServer.tool('analyze-data', schema, async (args) => {
  const result = await performAnalysis(args);
  return result; // Automatically summarized if content is long
});
